637 try: 

638 file_age = get_age(filepath, use_ctime=use_ctime) 

639 except ValueError: # pragma: no cover 

640 logger.warning('Could not get age of file %r in %r', filename, dirpath) 

641 continue 

642 else: 

643 if age and file_age > age: 

644 logger.debug('Skipping old file %r in %r', filename, dirpath) 

645 continue 

646 

647 # scan 

648 if filename.lower().endswith(VIDEO_EXTENSIONS): # video 

649 try: 

650 video = scan_video(filepath, name=name) 

651 except ValueError: # pragma: no cover 

652 logger.exception('Error scanning video') 

653 continue 

654 elif archives and filename.lower().endswith(ARCHIVE_EXTENSIONS): # archive 

655 try: 

656 video = scan_archive(filepath, name=name) 

657 except (Error, NotRarFile, RarCannotExec, ValueError): # pragma: no cover 

658 logger.exception('Error scanning archive') 

659 continue 

660 else: # pragma: no cover 

661 msg = f'Unsupported file {filename!r}' 

662 raise ValueError(msg) 

663 

664 videos.append(video) 

665 

666 return videos

669def refine( 

670 video: Video, 

671 *, 

672 refiners: Sequence[str] | None = None, 

673 refiner_configs: Mapping[str, Any] | None = None, 

674 **kwargs: Any, 

675) -> Video: 

676 """Refine a video using :ref:`refiners`. 

677 

678 .. note:: 

679 

680 Exceptions raised in refiners are silently passed and logged. 

681 

682 :param video: the video to refine. 

683 :type video: :class:`~subliminal.video.Video` 

684 :param Sequence refiners: refiners to select. None defaults to all refiners. 

685 :param dict refiner_configs: refiner configuration as keyword arguments per refiner name to pass when 

686 calling the refine method 

687 :param kwargs: additional parameters for the :func:`~subliminal.refiners.refine` functions. 

688 

689 """ 

690 refiners = refiners if refiners is not None else default_refiners 

691 if isinstance(video, Movie): 

692 refiners = [r for r in refiners if r not in discarded_movie_refiners] 

693 if isinstance(video, Episode): 

694 refiners = [r for r in refiners if r not in discarded_episode_refiners] 

695 

696 for refiner in refiners: 

697 logger.info('Refining video with %s', refiner) 

698 try: 

699 refiner_manager[refiner].plugin(video, **dict((refiner_configs or {}).get(refiner, {}), **kwargs)) 

700 except Exception as e: # noqa: BLE001 

701 handle_exception(e, f'Failed to refine video {video.name!r}') 

702 return video

705def list_subtitles( 

706 videos: Set[Video], 

707 languages: Set[Language], 

708 *, 

709 pool_class: type[ProviderPool] = ProviderPool, 

710 **kwargs: Any, 

711) -> dict[Video, list[Subtitle]]: 

712 """List subtitles. 

713 

714 The `videos` must pass the `languages` check of :func:`check_video`. 

715 

716 :param videos: videos to list subtitles for. 

717 :type videos: set of :class:`~subliminal.video.Video` 

718 :param languages: languages to search for. 

719 :type languages: set of :class:`~babelfish.language.Language` 

720 :param pool_class: class to use as provider pool. 

721 :type pool_class: :class:`ProviderPool`, :class:`AsyncProviderPool` or similar 

722 :param kwargs: additional parameters for the provided `pool_class` constructor. 

723 :return: found subtitles per video. 

724 :rtype: dict of :class:`~subliminal.video.Video` to list of :class:`~subliminal.subtitle.Subtitle` 

725 

726 """ 

727 listed_subtitles: dict[Video, list[Subtitle]] = defaultdict(list) 

728 

729 # check videos 

730 checked_videos = [] 

731 for video in videos: 

732 if not check_video(video, languages=languages): 

733 logger.info('Skipping video %r', video) 

734 continue 

735 checked_videos.append(video) 

736 

737 # return immediately if no video passed the checks 

738 if not checked_videos: 

739 return listed_subtitles 

740 

741 # list subtitles 

742 with pool_class(**kwargs) as pool: 

743 for video in checked_videos: 

744 logger.info('Listing subtitles for %r', video) 

745 subtitles = pool.list_subtitles(video, languages - video.subtitle_languages) 

746 listed_subtitles[video].extend(subtitles) 

747 logger.info('Found %d subtitle(s)', len(subtitles)) 

748 

749 return listed_subtitles

773def download_best_subtitles( 

774 videos: Set[Video], 

775 languages: Set[Language], 

776 *, 

777 min_score: int = 0, 

778 hearing_impaired: bool = False, 

779 only_one: bool = False, 

780 compute_score: ComputeScore | None = None, 

781 pool_class: type[ProviderPool] = ProviderPool, 

782 **kwargs: Any, 

783) -> dict[Video, list[Subtitle]]: 

784 """List and download the best matching subtitles. 

785 

786 The `videos` must pass the `languages` and `undefined` (`only_one`) checks of :func:`check_video`. 

787 

788 :param videos: videos to download subtitles for. 

789 :type videos: set of :class:`~subliminal.video.Video` 

790 :param languages: languages to download. 

791 :type languages: set of :class:`~babelfish.language.Language` 

792 :param int min_score: minimum score for a subtitle to be downloaded. 

793 :param bool hearing_impaired: hearing impaired preference. 

794 :param bool only_one: download only one subtitle, not one per language. 

795 :param compute_score: function that takes `subtitle` and `video` as positional arguments, 

796 `hearing_impaired` as keyword argument and returns the score. 

797 :param pool_class: class to use as provider pool. 

798 :type pool_class: :class:`ProviderPool`, :class:`AsyncProviderPool` or similar 

799 :param kwargs: additional parameters for the provided `pool_class` constructor. 

800 :return: downloaded subtitles per video. 

801 :rtype: dict of :class:`~subliminal.video.Video` to list of :class:`~subliminal.subtitle.Subtitle` 

802 

803 """ 

804 downloaded_subtitles: dict[Video, list[Subtitle]] = defaultdict(list) 

805 

806 # check videos 

807 checked_videos = [] 

808 for video in videos: 

809 if not check_video(video, languages=languages, undefined=only_one): 

810 logger.info('Skipping video %r', video) 

811 continue 

812 checked_videos.append(video) 

813 

814 # return immediately if no video passed the checks 

815 if not checked_videos: 

816 return downloaded_subtitles 

817 

818 # download best subtitles 

819 with pool_class(**kwargs) as pool: 

820 for video in checked_videos: 

821 logger.info('Downloading best subtitles for %r', video) 

822 subtitles = pool.download_best_subtitles( 

823 pool.list_subtitles(video, languages - video.subtitle_languages), 

824 video, 

825 languages, 

826 min_score=min_score, 

827 hearing_impaired=hearing_impaired, 

828 only_one=only_one, 

829 compute_score=compute_score, 

830 ) 

831 logger.info('Downloaded %d subtitle(s)', len(subtitles)) 

832 downloaded_subtitles[video].extend(subtitles) 

833 

834 return downloaded_subtitles

837def save_subtitles( 

838 video: Video, 

839 subtitles: Sequence[Subtitle], 

840 *, 

841 single: bool = False, 

842 directory: str | os.PathLike | None = None, 

843 encoding: str | None = None, 

844 extension: str | None = None, 

845 language_type_suffix: bool = False, 

846 language_format: str = 'alpha2', 

847) -> list[Subtitle]: 

848 """Save subtitles on filesystem. 

849 

850 Subtitles are saved in the order of the list. If a subtitle with a language has already been saved, other subtitles 

851 with the same language are silently ignored. 

852 

853 The extension used is `.lang.srt` by default or `.srt` is `single` is `True`, with `lang` being the IETF code for 

854 the :attr:`~subliminal.subtitle.Subtitle.language` of the subtitle. 

855 

856 :param video: video of the subtitles. 

857 :type video: :class:`~subliminal.video.Video` 

858 :param subtitles: subtitles to save. 

859 :type subtitles: list of :class:`~subliminal.subtitle.Subtitle` 

860 :param bool single: save a single subtitle, default is to save one subtitle per language. 

861 :param str directory: path to directory where to save the subtitles, default is next to the video. 

862 :param str encoding: encoding in which to save the subtitles, default is to keep original encoding. 

863 :param (str | None) extension: the subtitle extension, default is to match to the subtitle format. 

864 :param bool language_type_suffix: add a suffix 'hi' or 'forced' if needed. Default to False. 

865 :param str language_format: format of the language suffix. Default to 'alpha2'. 

866 :return: the saved subtitles 

867 :rtype: list of :class:`~subliminal.subtitle.Subtitle` 

868 

869 """ 

870 saved_subtitles: list[Subtitle] = [] 

871 for subtitle in subtitles: 

872 # check content 

873 if not subtitle.content: 

874 logger.error('Skipping subtitle %r: no content', subtitle) 

875 continue 

876 

877 # check language 

878 if subtitle.language in {s.language for s in saved_subtitles}: 

879 logger.debug('Skipping subtitle %r: language already saved', subtitle) 

880 continue 

881 

882 # create subtitle path 

883 subtitle_path = subtitle.get_path( 

884 video, 

885 single=single, 

886 extension=extension, 

887 language_type_suffix=language_type_suffix, 

888 language_format=language_format, 

889 ) 

890 if directory is not None: 

891 subtitle_path = os.path.join(directory, os.path.split(subtitle_path)[1]) 

892 

893 # save content as is or in the specified encoding 

894 logger.info('Saving %r to %r', subtitle, subtitle_path) 

895 if encoding is None: 

896 with open(subtitle_path, 'wb') as f: 

897 f.write(subtitle.content) 

898 else: 

899 with open(subtitle_path, 'w', encoding=encoding) as f: 

900 f.write(subtitle.text) 

901 saved_subtitles.append(subtitle) 

902 

903 # check single 

904 if single: 

905 break 

906 

907 return saved_subtitles
