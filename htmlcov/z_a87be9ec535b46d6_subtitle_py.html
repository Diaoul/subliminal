229 def reencode(self, encoding: str = 'utf-8') -> bool: 

230 """Re-encode the subtitle raw content using the specified encoding. 

231 

232 :param str encoding: the new encoding of the raw content (default to 'utf-8'). 

233 :return: False if the encoding raised a UnicodeEncodeError error. 

234 :rtype: bool 

235 

236 """ 

237 # Compute self._text by calling the property 

238 text = self.text 

239 

240 # Text is empty, maybe because the content was not decoded. 

241 # Reencoding would erase the content, so return. 

242 if not text: 

243 return False 

244 

245 # Try re-encoding 

246 try: 

247 new_content = text.encode(encoding=encoding) 

248 except UnicodeEncodeError: 

249 logger.exception('Cannot encode text to bytes with encoding: %s', encoding) 

250 return False 

251 

252 # Save the new encoding and new raw content 

253 self.encoding = encoding 

254 self._content = new_content 

255 return True

269 def _check_is_valid(self, *, auto_fix_srt: bool = False) -> bool: 

270 """Check if a :attr:`text` is a valid SubRip format. 

271 

272 :return: whether or not the subtitle is valid. 

273 :rtype: bool 

274 

275 """ 

276 if not self.text: 

277 return False 

278 

279 # Try guessing the subtitle format 

280 if self.subtitle_format is None: 

281 guessed_format = get_subtitle_format(self.text, subtitle_format=self.subtitle_format, fps=self.fps) 

282 if guessed_format: 

283 # Keep the guessed format 

284 self.subtitle_format = guessed_format 

285 if self.subtitle_format != 'srt': 

286 # Do not check more if the format is not 'srt' 

287 return True 

288 

289 # Valid srt 

290 if self.subtitle_format == 'srt': 

291 try: 

292 parsed = self.parse_srt() 

293 except Exception: 

294 msg = 'srt parsing failed, subtitle is invalid' 

295 logger.exception(msg) 

296 else: 

297 if auto_fix_srt: 

298 self._text = parsed 

299 return True 

300 

301 return False

303 def parse_srt(self) -> str: 

304 """Text content parsed to a valid srt subtitle.""" 

305 return str(srt.compose(srt.parse(self.text)))

307 def guess_encoding(self) -> str | None: 

308 """Guess encoding using the language, falling back on chardet. 

309 

310 :return: the guessed encoding. 

311 :rtype: str 

312 

313 """ 

314 if not isinstance(self.content, bytes): 

315 return None 

316 logger.info('Guessing encoding for language %s', self.language) 

317 

318 # always try utf-8 first 

319 encodings = ['utf-8'] 

320 

321 # add UTF encodings matched by the BOM 

322 encodings.extend(find_encoding_with_bom(self.content)) 

323 

324 # add language-specific encodings 

325 # http://scratchpad.wikia.com/wiki/Character_Encoding_Recommendation_for_Languages 

326 

327 if self.language.alpha3 == 'zho': 

328 encodings.extend( 

329 ['cp936', 'gb2312', 'gbk', 'hz', 'iso2022_jp_2', 'cp950', 'big5hkscs', 'big5', 'gb18030', 'utf-16'] 

330 ) 

331 

332 elif self.language.alpha3 == 'jpn': 

333 encodings.extend( 

334 [ 

335 'shift-jis', 

336 'cp932', 

337 'euc_jp', 

338 'iso2022_jp', 

339 'iso2022_jp_1', 

340 'iso2022_jp_2', 

341 'iso2022_jp_2004', 

342 'iso2022_jp_3', 

343 'iso2022_jp_ext', 

344 ] 

345 ) 

346 

347 elif self.language.alpha3 == 'tha': 

348 encodings.extend(['tis-620', 'cp874']) 

349 

350 elif self.language.alpha3 in ('ara', 'fas', 'per'): 

351 encodings.extend(['windows-1256', 'utf-16', 'utf-16le', 'ascii', 'iso-8859-6']) 

352 

353 elif self.language.alpha3 == 'heb': 

354 encodings.extend(['windows-1255', 'iso-8859-8']) 

355 

356 elif self.language.alpha3 == 'tur': 

357 encodings.extend(['windows-1254', 'iso-8859-9', 'iso-8859-3']) 

358 

359 elif self.language.alpha3 in ('grc', 'gre', 'ell'): 

360 encodings.extend( 

361 ['windows-1253', 'cp1253', 'cp737', 'iso8859-7', 'cp875', 'cp869', 'iso2022_jp_2', 'mac_greek'] 

362 ) 

363 

364 elif self.language.alpha3 in ( 

365 'pol', 

366 'cze', 

367 'ces', 

368 'slk', 

369 'slo', 

370 'slv', 

371 'hun', 

372 'bos', 

373 'hbs', 

374 'hrv', 

375 'rsb', 

376 'ron', 

377 'rum', 

378 'sqi', 

379 'alb', 

380 ): 

381 encodings.extend(['windows-1250', 'iso-8859-2']) 

382 

383 if self.language.alpha3 == 'slv': 

384 encodings.extend(['iso-8859-4']) 

385 elif self.language.alpha3 in ('sqi', 'alb'): 

386 encodings.extend(['windows-1252', 'iso-8859-15', 'iso-8859-1', 'iso-8859-9']) 

387 

388 elif self.language.alpha3 in ('bul', 'srp', 'mkd', 'mac', 'rus', 'ukr'): 

389 if self.language.alpha3 in ('bul', 'mkd', 'mac', 'rus', 'ukr'): 

390 encodings.extend(['windows-1251', 'iso-8859-5']) 

391 

392 elif self.language.alpha3 == 'srp': 

393 if self.language.script == 'Latn': 

394 encodings.extend(['windows-1250', 'iso-8859-2']) 

395 elif self.language.script == 'Cyrl': 

396 encodings.extend(['windows-1251', 'iso-8859-5']) 

397 else: 

398 encodings.extend(['windows-1250', 'windows-1251', 'iso-8859-2', 'iso-8859-5']) 

399 

400 else: 

401 # Western European (windows-1252) / Northern European 

402 encodings.extend(['windows-1252', 'iso-8859-15', 'iso-8859-9', 'iso-8859-4', 'iso-8859-1']) 

403 

404 # try to decode 

405 logger.debug('Trying encodings %r', encodings) 

406 for encoding in encodings: 

407 try: 

408 decoded = self.content.decode(encoding) 

409 # remove whitespace other than spaces from the string 

410 # see https://docs.python.org/3/library/stdtypes.html#str.isprintable 

411 decoded = decoded.replace('\r', '').replace('\n', '').replace('\t', '') 

412 if not decoded.isprintable(): 

413 continue 

414 except UnicodeDecodeError: 

415 pass 

416 else: 

417 logger.info('Guessed encoding %s', encoding) 

418 return encoding 

419 

420 logger.warning('Could not guess encoding from language') 

421 

422 # fallback on chardet 

423 encoding_or_none = chardet.detect(self.content)['encoding'] 

424 logger.info('Chardet found encoding %s', encoding_or_none) 

425 

426 return encoding_or_none

497def get_subtitle_format( 

498 text: str, 

499 subtitle_format: str | None = None, 

500 fps: float | None = None, 

501) -> str | None: 

502 """Detect the subtitle format with `pysubs2`. 

503 

504 :param str text: the subtitle text. 

505 :param (str | None) subtitle_format: the expected subtitle_format, None for auto-detect. 

506 :param (str | None) fps: the framerate for framerate based subtitles. 

507 :return: the guessed format or None if not found. 

508 :rtype: str | None 

509 

510 """ 

511 try: 

512 obj = SSAFile.from_string(text, format_=subtitle_format, fps=fps) 

513 except UnknownFPSError: 

514 default_fps = 24 

515 return get_subtitle_format(text, subtitle_format=subtitle_format, fps=default_fps) 

516 except Exception: 

517 logger.exception('not a valid subtitle.') 

518 else: 

519 return str(obj.format) 

520 return None

523def get_subtitle_path(video_path: str | os.PathLike, language: Language | None = None, extension: str = '.srt') -> str: 

524 """Get the subtitle path using the `video_path` and `language`. 

525 

526 :param str video_path: path to the video. 

527 :param language: language of the subtitle to put in the path. 

528 :type language: :class:`~babelfish.language.Language` 

529 :param str extension: extension of the subtitle. 

530 :return: path of the subtitle. 

531 :rtype: str

533 """ 

534 subtitle_root = os.path.splitext(video_path)[0]
